# Instalasi Google Cloud SDK di WSL (Ubuntu/Debian)
sudo apt update && sudo apt upgrade -y
sudo apt install apt-transport-https ca-certificates gnupg curl -y

# 2. Import GPG key (metode resmi terbaru)
curl -fsSL https://packages.cloud.google.com/apt/doc/apt-key.gpg \
  | gpg --dearmor \
  | sudo tee /usr/share/keyrings/google-cloud.gpg > /dev/null

# 3 Tambahkan repository dengan signed-by
echo "deb [signed-by=/usr/share/keyrings/google-cloud.gpg] https://packages.cloud.google.com/apt cloud-sdk main" \
  | sudo tee /etc/apt/sources.list.d/google-cloud-sdk.list

# 4. Update & install
sudo apt update
sudo apt install google-cloud-sdk -y

# 5. Coba Jalankan 
gcloud init
